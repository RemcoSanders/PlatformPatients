def tokenize(t):
    text = t.lower()
    text = re.sub("\n"," ",text)
    text = re.sub("nbsp"," ",text)
    text = re.sub(r'<[^>]+>'," ",text) # remove all html markup
    text = re.sub('[^a-zèéeêëûüùôöòóœøîïíàáâäæãåA-Z0-9- \']', "", text)
    text = re.sub("nbsp"," ",text)
    text = re.sub(" hi "," hoi ",text)    #samenvoegen openingen
    text = re.sub("hallo","hoi ",text) #samenvoegen openingen
    text = re.sub("hoij","hoi ",text) #samenvoegen openingen
    text = re.sub("hoier","hoi ",text) #samenvoegen openingen
    text = re.sub(" groet "," groetjes ",text) #samenvoegen afsluitingen
    text = re.sub("vraagt","vragen",text) #samenvoegen werkwoorden
    text = re.sub("vroeg","vragen",text) #samenvoegen werkwoorden
    text = re.sub("gevraagd","vragen",text) #samenvoegen werkwoorden
    text = re.sub(" laat "," laten ",text) #samenvoegen werkwoorden
    text = re.sub(" liet "," laten ",text) #samenvoegen werkwoorden
    text = re.sub("heeft","hebben ",text) #samenvoegen werkwoorden
    text = re.sub("hebt","hebben ",text) #samenvoegen werkwoorden
    text = re.sub(" heb "," hebben ",text) #samenvoegen werkwoorden
    text = re.sub(" kan "," kunnen ",text) #samenvoegen werkwoorden
    text = re.sub("gehad"," hebben ",text) #samenvoegen werkwoorden
    text = re.sub(" kon "," kunnen ",text) #samenvoegen werkwoorden
    text = re.sub("gekund","kunnen ",text) #samenvoegen werkwoorden
    text = re.sub("leeft","leven ",text) #samenvoegen werkwoorden
    text = re.sub("leef","leven ",text) #samenvoegen werkwoorden
    text = re.sub("geleefd","leven ",text) #samenvoegen werkwoorden
    text = re.sub("week","weken ",text) #samenvoegen duur
    text = re.sub("jaar","jaren ",text) #samenvoegen werkwoorden
    # spelfouten
    text = re.sub("misschoien","misschien",text) # spelling correction
    text = re.sub("choirurg", "chirug", text)  # spelling correction
    text = re.sub('[\d-]', ' ',text)
    text = re.sub("'", " ", text)
    wrds = text.split()
    return wrds
